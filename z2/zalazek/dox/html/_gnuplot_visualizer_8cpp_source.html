136 /*
137  * Czyta plik, który zawiera współrzędne bryły wzorocej. Przeczytane współrzędne
138  * wierzchołków są transformowane, tzn. podane są:
139  * \li rotacji,
140  * \li skalowaniu,
141  * \li translacji.
142  * Następnie zapisywane są do pliku, który będzie czytał gnuplot. Ważne jest,
143  * że funkcja ta zachowuje budowę pliku wzorcowego, tzn. w odpowiednim miejscu
144  * znajdzie się wolna linia między poszczególnymi zestawami współrzędnych
145  * wierzchołków.
146  *
147  * \param[in] sFile_Template - nazwa pliku zawierającego współrzędne wierzchołków bryły
148  * wzorcowej. Z tego pliku wspomniane współrzędne są czytane.
149  * \param[in] sFile_Target - nazwa pliku, do którego zostaną zapisane przetransformowane
150  * współrzędne.
151  * \param[in] rTrans_m - wektor translacji, którego współrzędne wyrażone są w metrach.
152  * \param[in] RotAngle_rad - kąt rotacji wyrażony w radianach.
153  *
154  * \retval true - gdy operacja otwarcia i zapisu odpowiednich plików zakończyła się
155  * powodzeniem,
156  * \retval false - w przypadku przeciwnym.
157  */
158 bool TransformGeom(
159  const char *sFile_Template,
160  const char *sFile_Target,
161  const Wektor3D &rTrans_m,
162  double RotAngle_rad,
163  const Wektor3D &rScale
164  )
165 {
166  ofstream OuStrm(sFile_Target);
167  ifstream InStrm(sFile_Template);
168  Wektor3D Pos;
169 
170  if (!OuStrm.is_open()) {
171  cerr << " Blad otwarcia do zapisu pliku \"" << sFile_Target << "\""
172  << endl;
173  return false;
174  }
175  if (!InStrm.is_open()) {
176  cerr << " Blad otwarcia do odczytu pliku \"" << sFile_Template << "\""
177  << endl;
178  return false;
179  }
180 
181  char Separator;
182  double sn = sin(RotAngle_rad), cn = cos(RotAngle_rad);
183 
184  while (!InStrm.eof()) {
185  if (InStrm >> Pos) {
186  // Najpierw dokonujemy rotacji
187  Pos = RotateXY(sn,cn,Pos,rScale);
188  // Następnie translacja
189  Pos += rTrans_m;
190  OuStrm << Pos << endl;
191  continue;
192  }
193  InStrm.clear();
194  if (!(InStrm >> Separator)) break;
195  if (Separator != '#') {
196  cerr << " Blad w pliku \"" << sFile_Target << "\"" << endl
197  << " oczekiwano znaku #. Zamiast tego odebrano znak " << Separator
198  << endl;
199  return false;
200  }
201  OuStrm << "#\n\n";
202  }
203  return true;
204 }

225 bool WriteCurrDonPose( const DronPose *pPose )

228  static double RotorAngle_rad = 0;
229  Wektor3D RotorScale(18,18,10);

237  Wektor3D Trans_Rot2(-DronScale.x(),DronScale.y(),DronScale.z());

239  Wektor3D Trans_Rot4(-DronScale.x(),-DronScale.y(),DronScale.z());
240 
241  RotorAngle_rad += 5*M_PI/180;
242 
243  if (!TransformGeom(FILE_NAME__DRON_BODY_TEMPLATE,FILE_NAME__DRON_BODY_VIEW,

246

262

269
